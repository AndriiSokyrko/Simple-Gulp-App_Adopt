How to install Gulp 4 before it's officially released
node version 11.9.0 (msi for windows)
npm rm gulp -g
npm rm gulp-cli -g
cd [your-project-dir/]
npm rm gulp --save-dev
npm rm gulp --save
npm rm gulp --save-optional npm cache clean

# Install the latest Gulp CLI tools globally
npm install gulp-cli -g

# Install Gulp 4 into your project as dev dependency
npm install gulp --save-dev

# Check the versions installed. Make sure your versions are not lower than shown.
gulp -v
# install from зфслуоуюоыщт
npm install
